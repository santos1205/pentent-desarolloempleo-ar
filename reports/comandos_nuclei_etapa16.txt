# ============================================
# ETAPA 16: AUTOMATED VULNERABILITY SCANNING
# Comandos Nuclei para executar manualmente
# ============================================

# Comando 1: Scan completo (todos os templates)
/c/Sec/Tools/nuclei/nuclei.exe -u https://desarrolloyempleo.cba.gov.ar -o etapa16_nuclei_full.txt

# Comando 2: Scan focado em exposições e configurações incorretas
/c/Sec/Tools/nuclei/nuclei.exe -u https://desarrolloyempleo.cba.gov.ar -t http/exposures,http/misconfiguration -o etapa16_nuclei_exposures.txt

# Comando 3: Scan focado em WordPress (já identificado)
/c/Sec/Tools/nuclei/nuclei.exe -u https://desarrolloyempleo.cba.gov.ar -tags wordpress -o etapa16_nuclei_wordpress.txt

# Comando 4: Scan de security headers
/c/Sec/Tools/nuclei/nuclei.exe -u https://desarrolloyempleo.cba.gov.ar -t http/security-headers -o etapa16_nuclei_headers.txt

# Comando 5: Scan de vulnerabilidades conhecidas (CVEs)
/c/Sec/Tools/nuclei/nuclei.exe -u https://desarrolloyempleo.cba.gov.ar -tags cve -o etapa16_nuclei_cves.txt

# Comando 6: Scan de tecnologias detectadas (AWS, CloudFront)
/c/Sec/Tools/nuclei/nuclei.exe -u https://desarrolloyempleo.cba.gov.ar -tags aws,cloudfront -o etapa16_nuclei_aws.txt

# Comando 7: Scan silencioso (apenas resultados, sem cores)
/c/Sec/Tools/nuclei/nuclei.exe -u https://desarrolloyempleo.cba.gov.ar -silent -o etapa16_nuclei_silent.txt

# Comando 8: Scan com severidade alta e crítica
/c/Sec/Tools/nuclei/nuclei.exe -u https://desarrolloyempleo.cba.gov.ar -severity high,critical -o etapa16_nuclei_high_critical.txt
